	\begin{lstlisting}
   	int grade;
	if(grade < 4)
		return "fail";
	else if (grade < 5)
		return "pass";
	else if(grade < 6)
		return "good";
	else if (grade == 6)
		return "excellent";
	else
		return "grade is not correct";
\end{lstlisting}
